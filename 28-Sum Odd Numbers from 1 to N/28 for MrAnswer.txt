#include <iostream>
using namespace std;

enum enOddorEven { Odd = 1, Even = 2 };

int ReadNumber() {
	int num;
	cout << "Please enter number : ";
	cin >> num;
	return num;
}


enOddorEven CheckNumberOddorEven(int num) {
	if (num % 2 != 0)
		return enOddorEven::Odd;
	else
		return enOddorEven::Even;
}

int UsingForLoop(int num) {

	cout << "\nUsing For Statements\n";
	cout << "************************\n";
	int sum = 0;

	for (int counter = 1; counter <= num; counter++)
	{
		if (CheckNumberOddorEven(counter) == enOddorEven::Odd)
		{
			sum += counter;
		}
	}

	return sum;
}

int main() {

	int num = ReadNumber();
	cout << UsingForLoop(num) << endl ;

	return 0;
}
